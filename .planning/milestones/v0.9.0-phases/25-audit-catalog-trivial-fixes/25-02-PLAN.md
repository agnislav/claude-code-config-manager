**TRIV-01: Sandbox section count in `getSectionItemCount()`**

Replace the `case SectionType.Sandbox: return '';` with count logic that mirrors `buildSandboxProperties()` flattening:

```typescript
case SectionType.Sandbox: {
  const s = scopedConfig.config.sandbox;
  if (!s) return '0 properties';
  let count = 0;
  for (const [key, value] of Object.entries(s)) {
    if (key === 'network' && typeof value === 'object' && value !== null) {
      count += Object.keys(value).length;
    } else {
      count++;
    }
  }
  return `${count} ${count !== 1 ? 'properties' : 'property'}`;
}
```
